import { graphql } from 'gatsby'
import Boxes from '../components/Boxes'

export const pageQuery = graphql`
  query {
    allMdx(
      filter: { frontmatter: { title: { ne: "" } } }
      sort: { fields: frontmatter___date, order: DESC }
    ) {
      edges {
        node {
          fields {
            slug
          }
          frontmatter {
            title
            date
            description
          }
        }
      }
    }
  }
`

<Boxes data={props.data.allMdx} />
